```ts
import type { DataSource, InternalRow, RowKey } from '../types/index.js'

export interface NormalizeResult<T> {
  rows: InternalRow<T>[]
  byKey: Map<RowKey, InternalRow<T>>
  childrenMap: Map<RowKey | null, RowKey[]>
}

function getKey<T>(row: T, rowKey: DataSource<T>['rowKey']): RowKey {
  if (typeof rowKey === 'function') return rowKey(row)
  const value = row[rowKey]
  if (typeof value !== 'string' && typeof value !== 'number') {
    throw new Error(`rowKey must resolve to string | number, got ${typeof value}`)
  }
  return value
}

export function normalize<T>(source: DataSource<T>): NormalizeResult<T> {
  const { data, rowKey, tree } = source

  if (!tree) {
    // Flat list mode: no tree
    const rows: InternalRow<T>[] = data.map((raw, index) => ({
      key: getKey(raw, rowKey),
      raw,
      parentKey: null,
      depth: 0,
      index,
      hasChildren: false,
      childrenLoaded: true,
    }))
    const byKey = new Map<RowKey, InternalRow<T>>()
    for (const row of rows) byKey.set(row.key, row)
    const childrenMap = new Map<RowKey | null, RowKey[]>([[null, rows.map((r) => r.key)]])
    return { rows, byKey, childrenMap }
  }

  throw new Error('tree modes not implemented yet')
}
```

```ts
import type { DataSource, InternalRow, RowKey } from '../types/index.js'

export interface NormalizeResult<T> {
  rows: InternalRow<T>[]
  byKey: Map<RowKey, InternalRow<T>>
  childrenMap: Map<RowKey | null, RowKey[]>
}

function getKey<T>(row: T, rowKey: DataSource<T>['rowKey']): RowKey {
  if (typeof rowKey === 'function') return rowKey(row)
  const value = row[rowKey]
  if (typeof value !== 'string' && typeof value !== 'number') {
    throw new Error(`rowKey must resolve to string | number, got ${typeof value}`)
  }
  return value
}

export function normalize<T>(source: DataSource<T>): NormalizeResult<T> {
  const { data, rowKey, tree } = source

  const rows: InternalRow<T>[] = []
  const byKey = new Map<RowKey, InternalRow<T>>()
  const childrenMap = new Map<RowKey | null, RowKey[]>()

  if (!tree) {
    for (let index = 0; index < data.length; index++) {
      const raw = data[index] as T
      const row: InternalRow<T> = {
        key: getKey(raw, rowKey),
        raw,
        parentKey: null,
        depth: 0,
        index,
        hasChildren: false,
        childrenLoaded: true,
      }
      rows.push(row)
      byKey.set(row.key, row)
    }
    childrenMap.set(null, rows.map((r) => r.key))
    return { rows, byKey, childrenMap }
  }

  if (tree.mode === 'children') {
    const { childrenKey } = tree
    childrenMap.set(null, [])

    const walk = (nodes: readonly T[], parentKey: RowKey | null, depth: number): void => {
      const siblingKeys: RowKey[] = []
      for (let index = 0; index < nodes.length; index++) {
        const raw = nodes[index] as T
        const key = getKey(raw, rowKey)
        const rawChildren = raw[childrenKey] as unknown
        const hasChildren = Array.isArray(rawChildren) && rawChildren.length > 0

        const row: InternalRow<T> = {
          key,
          raw,
          parentKey,
          depth,
          index,
          hasChildren,
          childrenLoaded: true,
        }
        rows.push(row)
        byKey.set(key, row)
        siblingKeys.push(key)

        if (hasChildren) {
          walk(rawChildren as readonly T[], key, depth + 1)
        }
      }
      childrenMap.set(parentKey, siblingKeys)
    }

    walk(data, null, 0)
    return { rows, byKey, childrenMap }
  }

  // parent mode — implemented in next task
  throw new Error('parent mode not implemented yet')
}
```

```ts
  if (tree.mode === 'parent') {
    const { parentKey: parentField } = tree

    // 第一遍：收集 raw → key + 建 childrenMap
    const allKeys: RowKey[] = []
    const rawByKey = new Map<RowKey, T>()
    const childrenOf = new Map<RowKey | null, RowKey[]>()
    childrenOf.set(null, [])

    for (const raw of data) {
      const key = getKey(raw, rowKey)
      allKeys.push(key)
      rawByKey.set(key, raw)
    }

    for (const raw of data) {
      const key = getKey(raw, rowKey)
      const rawParent = raw[parentField] as unknown
      const parent =
        rawParent === null || rawParent === undefined
          ? null
          : rawByKey.has(rawParent as RowKey)
            ? (rawParent as RowKey)
            : null // orphan → root
      const list = childrenOf.get(parent) ?? []
      list.push(key)
      childrenOf.set(parent, list)
    }

    // 第二遍：DFS 构建 rows，保证父在子前
    const walk = (parentKey: RowKey | null, depth: number): void => {
      const siblings = childrenOf.get(parentKey) ?? []
      for (let index = 0; index < siblings.length; index++) {
        const key = siblings[index] as RowKey
        const raw = rawByKey.get(key) as T
        const kids = childrenOf.get(key) ?? []
        const row: InternalRow<T> = {
          key,
          raw,
          parentKey,
          depth,
          index,
          hasChildren: kids.length > 0,
          childrenLoaded: true,
        }
        rows.push(row)
        byKey.set(key, row)
        if (kids.length > 0) walk(key, depth + 1)
      }
    }
    walk(null, 0)

    for (const [k, v] of childrenOf) childrenMap.set(k, v)

    return { rows, byKey, childrenMap }
  }

  throw new Error(`unknown tree mode`)
```

```ts
import type { RowKey } from '../types/index.js'
import type { DataModel } from './DataModel.js'

export function getDescendants<T>(model: DataModel<T>, ancestor: RowKey): RowKey[] {
  const out: RowKey[] = []
  const stack: RowKey[] = [...(model.childrenMap.get(ancestor) ?? [])]
  while (stack.length > 0) {
    const key = stack.shift() as RowKey
    out.push(key)
    const kids = model.childrenMap.get(key)
    if (kids && kids.length > 0) stack.unshift(...kids)
  }
  return out
}

export function isDescendant<T>(
  model: DataModel<T>,
  ancestor: RowKey | null,
  target: RowKey,
): boolean {
  if (ancestor === target) return false
  let current = model.byKey.get(target)
  while (current) {
    if (current.parentKey === ancestor) return true
    if (current.parentKey === null) return ancestor === null
    current = model.byKey.get(current.parentKey)
  }
  return false
}
```

```ts
import type { SortState } from '../types/index.js'

export function toggleSort(
  current: readonly SortState[],
  columnKey: string,
  options: { multi: boolean },
): SortState[] {
  const existingIdx = current.findIndex((s) => s.columnKey === columnKey)

  if (existingIdx === -1) {
    if (options.multi) {
      return [...current, { columnKey, direction: 'asc', priority: current.length }]
    }
    return [{ columnKey, direction: 'asc' }]
  }

  const existing = current[existingIdx] as SortState
  if (existing.direction === 'asc') {
    const next = [...current]
    next[existingIdx] = { ...existing, direction: 'desc' }
    return next
  }
  // desc → remove
  const next = current.filter((_, i) => i !== existingIdx)
  return options.multi ? next.map((s, i) => ({ ...s, priority: i })) : next
}
```

```ts
import type { RowKey } from '../types/index.js'
import type { DataModel } from '../data-model/DataModel.js'
import { getDescendants } from '../data-model/traverse.js'

export function toggleSelect(
  selected: ReadonlySet<RowKey>,
  key: RowKey,
): Set<RowKey> {
  const next = new Set(selected)
  if (next.has(key)) next.delete(key)
  else next.add(key)
  return next
}

export function cascadeSelect<T>(
  model: DataModel<T>,
  selected: ReadonlySet<RowKey>,
  key: RowKey,
  check: boolean,
): Set<RowKey> {
  const next = new Set(selected)
  const descendants = getDescendants(model, key)

  if (check) {
    next.add(key)
    for (const d of descendants) next.add(d)

    // walk ancestors: if all siblings selected, promote parent
    let cursor = model.get(key)?.parentKey ?? null
    while (cursor !== null) {
      const siblings = model.childrenMap.get(cursor) ?? []
      if (siblings.every((s) => next.has(s))) {
        next.add(cursor)
        cursor = model.get(cursor)?.parentKey ?? null
      } else break
    }
  } else {
    next.delete(key)
    for (const d of descendants) next.delete(d)

    // unselect all ancestors (they can no longer be fully-checked)
    let cursor = model.get(key)?.parentKey ?? null
    while (cursor !== null) {
      next.delete(cursor)
      cursor = model.get(cursor)?.parentKey ?? null
    }
  }
  return next
}

export type CheckboxState = 'checked' | 'unchecked' | 'indeterminate'

export function computeCheckboxState<T>(
  model: DataModel<T>,
  selected: ReadonlySet<RowKey>,
  key: RowKey,
): CheckboxState {
  if (selected.has(key)) return 'checked'
  const descendants = getDescendants(model, key)
  const someSelected = descendants.some((d) => selected.has(d))
  return someSelected ? 'indeterminate' : 'unchecked'
}
```

```ts
export function memoize<Args extends unknown[], Ret>(
  fn: (...args: Args) => Ret,
): (...args: Args) => Ret {
  let cache: { args: Args; result: Ret } | undefined
  return (...args: Args): Ret => {
    if (cache && cache.args.length === args.length && cache.args.every((a, i) => a === args[i])) {
      return cache.result
    }
    const result = fn(...args)
    cache = { args, result }
    return result
  }
}
```

```ts
import type {
  FilterState,
  FilterValue,
  InternalRow,
  RowKey,
  SortState,
} from '../types/index.js'
import type { DataModel } from '../data-model/DataModel.js'

export interface ComputeVisibleRowsAccessors<T> {
  getSortValue?: (row: T, columnKey: string) => unknown
  getFilterValue?: (row: T, columnKey: string) => unknown
  sorters?: Record<string, (a: T, b: T) => number>
}

function defaultCompare(a: unknown, b: unknown): number {
  if (a === b) return 0
  if (a === undefined || a === null) return -1
  if (b === undefined || b === null) return 1
  if (typeof a === 'number' && typeof b === 'number') return a - b
  return String(a).localeCompare(String(b))
}

function matchFilter(value: unknown, filter: NonNullable<FilterValue>): boolean {
  if (filter.type === 'enum') return filter.values.includes(value)
  if (filter.type === 'text') {
    const str = value === undefined || value === null ? '' : String(value)
    const needle = filter.value
    if (filter.predicate === 'contains') return str.includes(needle)
    if (filter.predicate === 'startsWith') return str.startsWith(needle)
    if (filter.predicate === 'equals') return str === needle
    return true
  }
  return true // custom → assume upstream已过滤
}

export function computeVisibleRows<T>(
  model: DataModel<T>,
  expanded: ReadonlySet<RowKey>,
  sort: readonly SortState[],
  filter: Readonly<FilterState>,
  accessors: ComputeVisibleRowsAccessors<T> = {},
): InternalRow<T>[] {
  const filterKeys = Object.keys(filter).filter((k) => filter[k])
  const passesFilter = (row: InternalRow<T>): boolean => {
    if (filterKeys.length === 0) return true
    const getValue = accessors.getFilterValue
    if (!getValue) return true
    return filterKeys.every((k) => {
      const value = getValue(row.raw, k)
      const f = filter[k]
      return f ? matchFilter(value, f) : true
    })
  }

  const rank = new Map<RowKey, InternalRow<T>>()
  for (const r of model.rows) rank.set(r.key, r)

  const compareSiblings = (a: InternalRow<T>, b: InternalRow<T>): number => {
    if (sort.length === 0) return a.index - b.index
    for (const s of sort) {
      let cmp = 0
      const customSorter = accessors.sorters?.[s.columnKey]
      if (customSorter) cmp = customSorter(a.raw, b.raw)
      else {
        const getValue = accessors.getSortValue
        cmp = getValue ? defaultCompare(getValue(a.raw, s.columnKey), getValue(b.raw, s.columnKey)) : 0
      }
      if (cmp !== 0) return s.direction === 'asc' ? cmp : -cmp
    }
    return a.index - b.index
  }

  const out: InternalRow<T>[] = []
  const walk = (parentKey: RowKey | null): void => {
    const siblingKeys = model.childrenMap.get(parentKey) ?? []
    const siblings = siblingKeys
      .map((k) => rank.get(k) as InternalRow<T>)
      .filter(passesFilter)
      .slice()
      .sort(compareSiblings)

    for (const row of siblings) {
      out.push(row)
      if (row.hasChildren && expanded.has(row.key)) walk(row.key)
    }
  }
  walk(null)
  return out
}
```

```ts
export interface PrefixSum {
  offsets: number[]
  total: number
}

export function buildPrefixSum(heights: readonly number[]): PrefixSum {
  const offsets: number[] = new Array(heights.length)
  let acc = 0
  for (let i = 0; i < heights.length; i++) {
    offsets[i] = acc
    acc += heights[i] as number
  }
  return { offsets, total: acc }
}

// binary search: largest offsets[i] <= scrollTop
export function findIndexAt(offsets: readonly number[], scrollTop: number): number {
  if (offsets.length === 0) return 0
  if (scrollTop <= 0) return 0

  let lo = 0
  let hi = offsets.length - 1
  while (lo < hi) {
    const mid = (lo + hi + 1) >>> 1
    if ((offsets[mid] as number) <= scrollTop) lo = mid
    else hi = mid - 1
  }
  return lo
}
```

```ts
import { buildPrefixSum, findIndexAt } from './prefixSum.js'

export interface ComputeRangeParams {
  scrollTop: number
  viewportHeight: number
  rowHeights: readonly number[]
  overscan: number
}

export interface ComputeRangeResult {
  startIndex: number
  endIndex: number
  totalHeight: number
  offsets: readonly number[]
}

export function computeVisibleRange({
  scrollTop,
  viewportHeight,
  rowHeights,
  overscan,
}: ComputeRangeParams): ComputeRangeResult {
  if (rowHeights.length === 0) {
    return { startIndex: 0, endIndex: -1, totalHeight: 0, offsets: [] }
  }

  const { offsets, total } = buildPrefixSum(rowHeights)

  const rawStart = findIndexAt(offsets, scrollTop)
  const bottomScroll = scrollTop + viewportHeight
  const rawEnd = findIndexAt(offsets, bottomScroll)

  const start = Math.max(0, rawStart - overscan)
  const end = Math.min(rowHeights.length - 1, rawEnd + overscan)

  return { startIndex: start, endIndex: end, totalHeight: total, offsets }
}
```

```ts
export interface LayoutColumn {
  key: string
  width?: number | 'auto' | `${number}%`
  minWidth?: number
  maxWidth?: number
  flex?: number
}

export interface LayoutOptions {
  totalWidth: number
}

const DEFAULT_MIN = 40

export function distributeWidths(
  columns: readonly LayoutColumn[],
  { totalWidth }: LayoutOptions,
): Record<string, number> {
  const out: Record<string, number> = {}
  let fixedTotal = 0
  const flexEntries: LayoutColumn[] = []

  for (const col of columns) {
    if (typeof col.width === 'number') {
      const w = clamp(col.width, col.minWidth ?? DEFAULT_MIN, col.maxWidth)
      out[col.key] = w
      fixedTotal += w
    } else if (typeof col.width === 'string' && col.width.endsWith('%')) {
      const percent = parseFloat(col.width.slice(0, -1))
      const w = clamp((percent / 100) * totalWidth, col.minWidth ?? DEFAULT_MIN, col.maxWidth)
      out[col.key] = w
      fixedTotal += w
    } else {
      flexEntries.push(col)
    }
  }

  if (flexEntries.length === 0) return out

  const remaining = Math.max(0, totalWidth - fixedTotal)
  const totalFlex = flexEntries.reduce((sum, c) => sum + (c.flex ?? 1), 0) || flexEntries.length

  for (const col of flexEntries) {
    const share = ((col.flex ?? 1) / totalFlex) * remaining
    out[col.key] = clamp(share, col.minWidth ?? DEFAULT_MIN, col.maxWidth)
  }

  return out
}

function clamp(value: number, min: number, max?: number): number {
  if (max !== undefined && value > max) value = max
  if (value < min) value = min
  return Math.round(value)
}
```

```ts
const DEFAULT_MIN = 40
const DEFAULT_MAX = 9999

export interface ResolveOptions {
  totalWidth: number
}

export function resolveColumns<T>(
  columns: readonly ColumnDef<T>[],
  options: ResolveOptions,
): ResolvedColumn<T>[] {
  const leaves: (ColumnDef<T> & { depth: number })[] = []
  const collect = (list: readonly ColumnDef<T>[], depth: number): void => {
    for (const col of list) {
      if (col.children && col.children.length > 0) collect(col.children, depth + 1)
      else leaves.push({ ...col, depth })
    }
  }
  collect(columns, 0)

  const visible = leaves.filter((c) => !c.hidden)
  const widths = distributeWidths(visible, options)

  return visible.map((col) => ({
    key: col.key,
    title: col.title,
    field: col.field,
    render: col.render,
    computedWidth: widths[col.key] ?? DEFAULT_MIN,
    minWidth: col.minWidth ?? DEFAULT_MIN,
    maxWidth: col.maxWidth ?? DEFAULT_MAX,
    fixed: col.fixed,
    hidden: false,
    sortable: col.sortable === true || typeof col.sortable === 'object',
    sortMulti: typeof col.sortable === 'object' && col.sortable.multi === true,
    sorter: col.sorter,
    filter: col.filter,
    depth: col.depth,
    ariaLabel: col.ariaLabel,
  }))
}
```

```ts
import type { RowKey } from '../types/index.js'
import type { DataModel } from '../data-model/DataModel.js'

// True if `candidate` is `dragKey` itself, or a descendant of `dragKey`.
// Dropping into such target would create a cycle.
export function wouldCreateCycle<T>(
  model: DataModel<T>,
  dragKey: RowKey,
  candidateParent: RowKey | null,
): boolean {
  if (candidateParent === null) return false
  if (candidateParent === dragKey) return true
  let cursor = model.get(candidateParent)
  while (cursor) {
    if (cursor.key === dragKey) return true
    if (cursor.parentKey === null) return false
    cursor = model.get(cursor.parentKey)
  }
  return false
}
```

```ts
import type { DropCtx, RowKey } from '../types/index.js'
import type { DataModel } from '../data-model/DataModel.js'
import { wouldCreateCycle } from './validate.js'

export interface HitTestParams<T> {
  dragKey: RowKey
  pointerY: number
  targetKey: RowKey
  targetRect: { top: number; height: number }
  model: DataModel<T>
  expandedKeys: ReadonlySet<RowKey>
  allowInsideLeaf: boolean
}

export function computeDropTarget<T>(params: HitTestParams<T>): DropCtx<T> | null {
  const { dragKey, pointerY, targetKey, targetRect, model, expandedKeys, allowInsideLeaf } = params

  const target = model.get(targetKey)
  if (!target) return null
  const dragRow = model.get(dragKey)
  if (!dragRow) return null

  const ratio = (pointerY - targetRect.top) / targetRect.height
  const canInside = target.hasChildren || allowInsideLeaf

  let position: 'before' | 'after' | 'inside'
  let newParentKey: RowKey | null
  let newIndex: number
  let highlight: { row: RowKey; style: 'inside' } | null

  if (ratio < 1 / 3) {
    position = 'before'
    newParentKey = target.parentKey
    newIndex = target.index
    highlight = target.parentKey !== null ? { row: target.parentKey, style: 'inside' } : null
  } else if (ratio < 2 / 3 && canInside) {
    position = 'inside'
    newParentKey = target.key
    newIndex = expandedKeys.has(target.key)
      ? 0
      : (model.childrenMap.get(target.key)?.length ?? 0)
    highlight = { row: target.key, style: 'inside' }
  } else {
    if (target.hasChildren && expandedKeys.has(target.key)) {
      position = 'inside'
      newParentKey = target.key
      newIndex = 0
      highlight = { row: target.key, style: 'inside' }
    } else {
      position = 'after'
      newParentKey = target.parentKey
      newIndex = target.index + 1
      highlight = target.parentKey !== null ? { row: target.parentKey, style: 'inside' } : null
    }
  }

  if (wouldCreateCycle(model, dragKey, newParentKey)) return null

  return {
    dragRow: dragRow.raw,
    dragKey,
    targetRow: target.raw,
    targetKey,
    position,
    newParentKey,
    newIndex,
    highlight,
  }
}
```

```ts
import type { DataSource, DropCtx, RowKey, TreeConfig } from '../types/index.js'

export interface ApplyDropConfig<T> {
  rowKey: DataSource<T>['rowKey']
  tree?: TreeConfig<T>
}

function getKey<T>(row: T, rowKey: DataSource<T>['rowKey']): RowKey {
  if (typeof rowKey === 'function') return rowKey(row)
  return row[rowKey] as RowKey
}

export function applyDrop<T>(
  data: readonly T[],
  ctx: DropCtx<T>,
  config: ApplyDropConfig<T>,
): T[] {
  const { rowKey, tree } = config

  if (!tree || tree.mode === 'parent') {
    return applyDropFlat(data, ctx, rowKey, tree?.mode === 'parent' ? tree.parentKey : undefined)
  }

  return applyDropNested(data, ctx, rowKey, tree.childrenKey)
}

function applyDropFlat<T>(
  data: readonly T[],
  ctx: DropCtx<T>,
  rowKey: DataSource<T>['rowKey'],
  parentField: keyof T | undefined,
): T[] {
  const result: T[] = []
  let dragged: T | null = null

  for (const row of data) {
    const key = getKey(row, rowKey)
    if (key === ctx.dragKey) {
      const patched = parentField
        ? ({ ...row, [parentField]: ctx.newParentKey } as T)
        : ({ ...row } as T)
      dragged = patched
      continue
    }
    result.push(row)
  }

  if (!dragged) return [...data]

  // Insert at newIndex among siblings sharing newParentKey.
  // We iterate result to find nth same-parent sibling.
  let insertAt = result.length
  let siblingCount = 0
  for (let i = 0; i < result.length; i++) {
    const rowParent = parentField ? (result[i] as T)[parentField] : null
    const matches = (rowParent ?? null) === ctx.newParentKey
    if (matches) {
      if (siblingCount === ctx.newIndex) {
        insertAt = i
        break
      }
      siblingCount++
    }
  }
  result.splice(insertAt, 0, dragged)
  return result
}

function applyDropNested<T>(
  data: readonly T[],
  ctx: DropCtx<T>,
  rowKey: DataSource<T>['rowKey'],
  childrenKey: keyof T,
): T[] {
  let dragged: T | null = null

  // Deep clone helper (structure only, not values)
  const extract = (list: readonly T[]): T[] => {
    const out: T[] = []
    for (const node of list) {
      const key = getKey(node, rowKey)
      if (key === ctx.dragKey) {
        dragged = node
        continue
      }
      const kids = node[childrenKey] as unknown
      if (Array.isArray(kids)) {
        out.push({ ...node, [childrenKey]: extract(kids as readonly T[]) } as T)
      } else {
        out.push({ ...node })
      }
    }
    return out
  }

  const stripped = extract(data)
  if (!dragged) return [...data]

  const insertInto = (list: T[], parentKey: RowKey | null): T[] => {
    if (parentKey === null) {
      const next = [...list]
      next.splice(ctx.newIndex, 0, dragged as T)
      return next
    }
    return list.map((node) => {
      const key = getKey(node, rowKey)
      if (key === parentKey) {
        const kids = ((node[childrenKey] as T[] | undefined) ?? []).slice()
        kids.splice(ctx.newIndex, 0, dragged as T)
        return { ...node, [childrenKey]: kids } as T
      }
      const kids = node[childrenKey] as unknown
      if (Array.isArray(kids)) {
        return { ...node, [childrenKey]: insertInto(kids as T[], parentKey) } as T
      }
      return node
    })
  }

  return insertInto(stripped, ctx.newParentKey)
}
```

```ts
import type { ResolvedColumn } from '../types/index.js'
import type { DataModel } from '../data-model/DataModel.js'

export interface CsvOptions<T = unknown> {
  columns?: string[]
  includeHiddenColumns?: boolean
  separator?: string
  newline?: string
  header?: boolean
  formatter?: (value: unknown, column: ResolvedColumn<T>, row: T) => string
}

function escapeCell(value: string, separator: string): string {
  const needsQuote =
    value.includes(separator) || value.includes('"') || value.includes('\n')
  if (!needsQuote) return value
  return `"${value.replace(/"/g, '""')}"`
}

export function toCsv<T>(
  model: DataModel<T>,
  columns: readonly ResolvedColumn<T>[],
  options: CsvOptions<T> = {},
): string {
  const {
    columns: keys,
    separator = ',',
    newline = '\n',
    header = true,
    formatter,
  } = options

  const selected = keys
    ? columns.filter((c) => keys.includes(c.key))
    : columns.filter((c) => !c.hidden)

  const lines: string[] = []
  if (header) {
    lines.push(selected.map((c) => escapeCell(String(c.key), separator)).join(separator))
  }

  for (const row of model.rows) {
    const cells = selected.map((col) => {
      const raw = col.field !== undefined ? row.raw[col.field] : undefined
      const str = formatter
        ? formatter(raw, col, row.raw)
        : raw === undefined || raw === null
          ? ''
          : String(raw)
      return escapeCell(str, separator)
    })
    lines.push(cells.join(separator))
  }

  return lines.join(newline)
}
```
