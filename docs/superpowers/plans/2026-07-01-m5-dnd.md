**Goal:** 行拖拽（同层 + 跨层 before/after/inside）+ 列拖拽排序 + 列宽 resize + 完整 UX（DragOverlay 悬浮预览 + 横线指示 + inside 高亮 + autoScroll + allowDrop/beforeDrop 校验 + 循环检测 + 虚拟滚动兼容）+ Playwright E2E 测试。

**Tech Stack:** @dnd-kit/core, @dnd-kit/sortable, @dnd-kit/modifiers, Playwright

```ts
import { useState, useRef } from 'react'
import {
  computeDropTarget,
  wouldCreateCycle,
  type DropCtx,
  type InternalRow,
  type RowKey,
} from '@draggable-table/core'
import type { DataModel } from '@draggable-table/core'

export interface UseDndOptions<T> {
  model: DataModel<T>
  expanded: ReadonlySet<RowKey>
  allowInsideLeaf: boolean
  allowDrop?: (ctx: DropCtx<T>) => boolean
  beforeDrop?: (ctx: DropCtx<T>) => Promise<boolean>
  onDragEnd?: (ctx: DropCtx<T>) => void
}

export interface DndState<T> {
  draggingKey: RowKey | null
  dropCtx: DropCtx<T> | null
  startDrag: (key: RowKey) => void
  updateDrop: (targetKey: RowKey, pointerY: number, targetRect: DOMRect) => void
  clearHover: () => void
  finishDrag: () => Promise<void>
  cancelDrag: () => void
}

export function useDnd<T>(options: UseDndOptions<T>): DndState<T> {
  const { model, expanded, allowInsideLeaf, allowDrop, beforeDrop, onDragEnd } = options
  const [draggingKey, setDraggingKey] = useState<RowKey | null>(null)
  const [dropCtx, setDropCtx] = useState<DropCtx<T> | null>(null)
  const inFlightRef = useRef(false)

  const startDrag = (key: RowKey): void => {
    setDraggingKey(key)
    setDropCtx(null)
  }

  const updateDrop = (targetKey: RowKey, pointerY: number, targetRect: DOMRect): void => {
    if (draggingKey === null) return
    const ctx = computeDropTarget<T>({
      dragKey: draggingKey,
      pointerY,
      targetKey,
      targetRect: { top: targetRect.top, height: targetRect.height },
      model,
      expandedKeys: expanded,
      allowInsideLeaf,
    })
    if (!ctx) {
      setDropCtx(null)
      return
    }
    if (wouldCreateCycle(model, draggingKey, ctx.newParentKey)) {
      setDropCtx(null)
      return
    }
    if (allowDrop && !allowDrop(ctx)) {
      setDropCtx(null)
      return
    }
    setDropCtx(ctx)
  }

  const clearHover = (): void => {
    setDropCtx(null)
  }

  const finishDrag = async (): Promise<void> => {
    if (!dropCtx || inFlightRef.current) {
      setDraggingKey(null)
      setDropCtx(null)
      return
    }
    if (beforeDrop) {
      inFlightRef.current = true
      try {
        const ok = await beforeDrop(dropCtx)
        if (!ok) {
          setDraggingKey(null)
          setDropCtx(null)
          return
        }
      } finally {
        inFlightRef.current = false
      }
    }
    onDragEnd?.(dropCtx)
    setDraggingKey(null)
    setDropCtx(null)
  }

  const cancelDrag = (): void => {
    setDraggingKey(null)
    setDropCtx(null)
  }

  return { draggingKey, dropCtx, startDrag, updateDrop, clearHover, finishDrag, cancelDrag }
}
```

```tsx
import { useRef } from 'react'
import type { InternalRow, ResolvedColumn, RowKey } from '@draggable-table/core'
import { Row } from './Row.js'

export interface DraggableRowProps<T> {
  row: InternalRow<T>
  columns: readonly ResolvedColumn<T>[]
  rowIndex: number
  expanded: boolean
  onToggleExpand: (key: RowKey) => void
  onDragStart: (key: RowKey, rect: DOMRect) => void
  onDragOver: (key: RowKey, pointerY: number, rect: DOMRect) => void
  onDragEnd: () => void
  isDragging: boolean
  style?: React.CSSProperties
}

export function DraggableRow<T>(props: DraggableRowProps<T>) {
  const {
    row,
    onDragStart,
    onDragOver,
    onDragEnd,
    isDragging,
    style,
    ...rest
  } = props

  const ref = useRef<HTMLDivElement | null>(null)
  const isDraggingRef = useRef(false)

  const handlePointerDown = (e: React.PointerEvent): void => {
    const target = e.target as HTMLElement
    if (!target.closest('[data-drag-handle]')) return
    if (!ref.current) return
    e.currentTarget.setPointerCapture(e.pointerId)
    isDraggingRef.current = true
    onDragStart(row.key, ref.current.getBoundingClientRect())
  }

  const handlePointerMove = (e: React.PointerEvent): void => {
    if (!isDraggingRef.current) return
    if (!ref.current) return
    onDragOver(row.key, e.clientY, ref.current.getBoundingClientRect())
  }

  const handlePointerUp = (e: React.PointerEvent): void => {
    if (!isDraggingRef.current) return
    isDraggingRef.current = false
    e.currentTarget.releasePointerCapture(e.pointerId)
    onDragEnd()
  }

  return (
    <div
      ref={ref}
      style={{ ...style, opacity: isDragging ? 0.4 : undefined }}
      onPointerDown={handlePointerDown}
      onPointerMove={handlePointerMove}
      onPointerUp={handlePointerUp}
      onPointerCancel={handlePointerUp}
    >
      <Row {...rest} row={row} />
    </div>
  )
}
```

Note: 我们**不用 dnd-kit 的 `DndContext` for rows**——因为虚拟滚动下 row DOM 不稳定，用原生 pointer events + core hitTest 是最直接的方案。dnd-kit 主要用在列拖拽（那边 DOM 稳定）。这符合 spec 里"pointer-based，drag state 与被拖 DOM 无耦合"的意图（我们的实现里 drag state 存 useDnd hook 里，被拖行只是标记 `isDragging`，其它 pointer 事件都发在 target 行上）。

- [ ] **Step 3: AutoScroll**

在 useDnd 里加 auto-scroll：当 `updateDrop` 的 pointerY 距 scrollRef 顶/底 40px 内时，用 requestAnimationFrame 循环 scrollBy。

`packages/table/src/hooks/useDnd.ts` 追加：

```ts
export function useAutoScroll(
  ref: React.RefObject<HTMLElement | null>,
  active: boolean,
  pointerY: number,
): void {
  const rafRef = useRef<number | null>(null)
  useEffect(() => {
    if (!active) {
      if (rafRef.current !== null) cancelAnimationFrame(rafRef.current)
      return
    }
    const step = (): void => {
      const el = ref.current
      if (!el) return
      const rect = el.getBoundingClientRect()
      const zone = 40
      const distTop = pointerY - rect.top
      const distBottom = rect.bottom - pointerY
      let dy = 0
      if (distTop < zone) dy = -((zone - distTop) / zone) * 20
      else if (distBottom < zone) dy = ((zone - distBottom) / zone) * 20
      if (dy !== 0) el.scrollTop += dy
      rafRef.current = requestAnimationFrame(step)
    }
    rafRef.current = requestAnimationFrame(step)
    return () => {
      if (rafRef.current !== null) cancelAnimationFrame(rafRef.current)
    }
  }, [ref, active, pointerY])
}
```

需要 import useEffect / useRef。在 Table 里跟踪 pointerY state 并传入：

```tsx
import { useEffect, useRef } from 'react'

export interface ResizeHandleProps {
  columnKey: string
  currentWidth: number
  minWidth: number
  maxWidth: number
  onChange: (columnKey: string, width: number) => void
}

export function ResizeHandle({ columnKey, currentWidth, minWidth, maxWidth, onChange }: ResizeHandleProps) {
  const startXRef = useRef(0)
  const startWidthRef = useRef(0)

  const handlePointerDown = (e: React.PointerEvent): void => {
    e.stopPropagation()
    e.currentTarget.setPointerCapture(e.pointerId)
    startXRef.current = e.clientX
    startWidthRef.current = currentWidth
  }

  const handlePointerMove = (e: React.PointerEvent): void => {
    if (!e.currentTarget.hasPointerCapture(e.pointerId)) return
    const dx = e.clientX - startXRef.current
    const w = Math.min(maxWidth, Math.max(minWidth, startWidthRef.current + dx))
    onChange(columnKey, w)
  }

  const handlePointerUp = (e: React.PointerEvent): void => {
    if (e.currentTarget.hasPointerCapture(e.pointerId)) {
      e.currentTarget.releasePointerCapture(e.pointerId)
    }
  }

  return (
    <span
      className="dt-resize-handle"
      onPointerDown={handlePointerDown}
      onPointerMove={handlePointerMove}
      onPointerUp={handlePointerUp}
      onPointerCancel={handlePointerUp}
      style={{
        cursor: 'col-resize',
        width: 6,
        alignSelf: 'stretch',
        marginLeft: 'auto',
        background: 'transparent',
      }}
    />
  )
}
```

```tsx
  children,
}: DraggableHeaderCellProps) {
  const isDraggingRef = useRef(false)

  return (
    <div
      draggable
      onDragStart={(e) => {
        isDraggingRef.current = true
        e.dataTransfer.setData('text/plain', columnKey)
        onDragStart()
      }}
      onDragOver={(e) => {
        if (!isDraggingRef.current) return
        e.preventDefault()
      }}
      onDrop={(e) => {
        const from = e.dataTransfer.getData('text/plain')
        onDragOver(from)
      }}
      onDragEnd={() => {
        isDraggingRef.current = false
        onDragEnd()
      }}
    >
      {children}
    </div>
  )
}
```

在 HeaderGroup 里用 DraggableHeaderCell 包 HeaderCell（若 `draggable.columns=true`）。
